async function prosesUpload(){ const file = document.getElementById("csvFile").files[0]; if(!file){ alert("Pilih file CSV terlebih dahulu"); return; } document.getElementById("status").innerHTML = "Membaca file..."; const reader = new FileReader(); reader.onload = async function(e){ try{ const text = e.target.result; // Memecah CSV dengan aman (handle koma di dalam tanda kutip) const rows = text.split(/\r?\n/) .filter(row => row.trim() !== "") .map(row => row.split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/)); document.getElementById("status").innerHTML = "Mengupload..."; const response = await fetch(APP_URL, { method: "POST", redirect: "follow", // PENTING: Untuk menangani redirect Apps Script body: JSON.stringify({ action: "uploadCSV", rows: rows }) }); const result = await response.json(); if(result.status === "success"){ document.getElementById("status").innerHTML = "✅ Upload berhasil"; alert("Data berhasil masuk ke RETUR_GLOBAL"); } else { throw new Error("Gagal di server"); } } catch(err){ console.error(err); document.getElementById("status").innerHTML = "❌ Terjadi kesalahan"; alert("Error: " + err.message); } }; reader.readAsText(file); }
